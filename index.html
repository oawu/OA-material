Javascript 寫法 
  $(function () {
    $('.oa-select').OAjellyDropdown ({
      color:       null,    // 顏色樣式，可用值參照顏色設定。  可用 html data-color 設定。         預設值: null
      width:       '100px', // 寬度。                      可用 html data-width 設定。         預設值: '100px'
      isRipple:    true,    // 是否使用水波紋效果。           可用 html data-isRipple 設定。     預設值: true
      autoOpen:    false,   // 是否自動展開選單。            可用 html data-autoOpen 設定。      預設值: false
      paddingLeft: '12px'   // 左邊 padding 的值。          可用 html data-paddingLeft 設定。   預設值: '12px'
    });
  });
